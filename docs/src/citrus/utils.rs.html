13/// The source function `S` is defined as `j_nu/alpha`, which is clearly not
14/// defined for `alpha==0`. However `S` is used in the algorithm only in the
15/// term `(1-exp[-alpha*ds])*S`, which is defined for all values of alpha.
16/// The present function calculates this term and returns it in the
17/// argument remnantSnu. For values of `abs(alpha*ds)` less than a pre-
18/// calculated cutoff supplied in `ConfigInfo`, a Taylor approximation is
19/// used.
20/// Note that the same cutoff condition holds for replacement of
21/// `exp(-dTau)` by its Taylor expansion to 3rd order.
22pub fn calc_source_fn(dtau: f64, taylor_cutoff: f64) -> (f64, f64) {
23    let remant_snu: f64;
24    let exp_dtau: f64;
25    if dtau.abs() < taylor_cutoff {
26        remant_snu = 1.0 - dtau * (1.0 - dtau * (1.0 / 3.0)) * (1.0 / 2.0);
27        exp_dtau = 1.0 - dtau * (remant_snu);
28    } else {
29        exp_dtau = (-dtau).exp();
30        remant_snu = (1.0 - exp_dtau) / dtau;
31    }
32
33    (remant_snu, exp_dtau)
34}

36pub fn planck_fn(freq: f64, t_kelvin: f64) -> f64 {
37    if t_kelvin < cc::CITRUS_GLOBAL_EPS {
38        0.0
39    } else {
40        let wn = freq / cc::SPEED_OF_LIGHT_SI;
41        if cc::PLANCK_SI * freq > 100.0 * cc::BOLTZMANN_SI * t_kelvin {
42            2.0 * cc::PLANCK_SI
43                * wn
44                * wn
45                * freq
46                * (-cc::PLANCK_SI * freq / cc::BOLTZMANN_SI / t_kelvin).exp()
47        } else {
48            2.0 * cc::PLANCK_SI * wn * wn * freq
49                / ((cc::PLANCK_SI * freq / cc::BOLTZMANN_SI / t_kelvin - 1.0).exp() - 1.0)
50        }
51    }
52}

58pub fn get_erf(x0: f64, x1: f64) -> f64 {
59    let val0 = erf(x0);
60    let val1 = erf(x1);
61    let diff = val1 - val0;
62    let delta_x = x1 - x0;
63    (diff / delta_x).abs()
64}

66/// This function:
67/// 1. Converts frequency to wavelength in log10 space
68/// 2. Handles extrapolation outside the table range
69/// 3. Converts results from cm²/g to m²/kg (0.1 factor)
70///
71/// Parameters:
72/// - freq: Frequency
73/// - lamtab: Array of log10 wavelength values
74/// - kaptab: Array of log10 opacity values
75/// - n_entries: Number of entries in the arrays
76pub fn interpolate_kappa(freq: f64, lam: &RVecView, kap: &RVecView) -> Result<f64, SplineError> {
77    if lam.len() != kap.len() {
78        return Err(SplineError::DifferentLengths);
79    }
80
81    // Create the spline
82    let spline = CubicSpline::new(lam, kap)?;
83
84    // Calculate log10 of wavelength (SPEED_OF_LIGHT_SI / freq)
85    let loglam = (cc::SPEED_OF_LIGHT_SI / freq).log10();
86
87    let n_entries = lam.len();
88    let kappa_log10: f64;
89
90    if loglam < lam[0] {
91        // Below range: linear extrapolation in log space
92        kappa_log10 = kap[0] + (loglam - lam[0]) * (kap[1] - kap[0]) / (lam[1] - lam[0]);
93    } else if loglam > lam[n_entries - 1] {
94        // Above range: linear extrapolation in log space
95        kappa_log10 = kap[n_entries - 2]
96            + (loglam - lam[n_entries - 2]) * (kap[n_entries - 1] - kap[n_entries - 2])
97                / (lam[n_entries - 1] - lam[n_entries - 2]);
98    } else {
99        // In range: use the spline
100        kappa_log10 = spline.interpolate(loglam)?;
101    }
102
103    // Convert from log10 opacity and apply unit conversion factor (cm²/g to m²/kg)
104    let kappa = 0.1 * 10f64.powf(kappa_log10);
105
106    Ok(kappa)
107}

125        cc::AMU_SI * gas_mass_density_amus / gtd

164    pub struct CubicSpline {
165        x: RVector,
166        y: RVector,
167        // Second derivatives at each point
168        y2: RVector,
169    }
170
171    impl CubicSpline {
172        /// Create a new cubic spline interpolator
173        pub fn new(x: &RVecView, y: &RVecView) -> Result<Self, SplineError> {
174            if x.len() != y.len() {
175                return Err(SplineError::DifferentLengths);
176            }
177            if x.len() < 2 {
178                return Err(SplineError::InsufficientPoints);
179            }
180
181            // Check that x is sorted in ascending order
182            for i in 1..x.len() {
183                if x[i] <= x[i - 1] {
184                    return Err(SplineError::UnsortedPoints);
185                }
186            }
187
188            let n = x.len();
189            let mut y2 = RVector::zeros(n);
190
191            // This computes the second derivatives needed for the spline
192            // Natural spline conditions: y''(x[0]) = y''(x[n-1]) = 0
193            let mut u = vec![0.0; n - 1];
194
195            // Decomposition loop of the tridiagonal algorithm
196            for i in 1..n - 1 {
197                let sig = (x[i] - x[i - 1]) / (x[i + 1] - x[i - 1]);
198                let p = sig * y2[i - 1] + 2.0;
199                // Check for numerical stability
200                if p.abs() < f64::EPSILON {
201                    return Err(SplineError::NumericalError);
202                }
203                y2[i] = (sig - 1.0) / p;
204                u[i] =
205                    (y[i + 1] - y[i]) / (x[i + 1] - x[i]) - (y[i] - y[i - 1]) / (x[i] - x[i - 1]);
206                u[i] = (6.0 * u[i] / (x[i + 1] - x[i - 1]) - sig * u[i - 1]) / p;
207            }
208
209            // Back-substitution loop of the tridiagonal algorithm
210            for i in (0..n - 1).rev() {
211                y2[i] = y2[i] * y2[i + 1] + u[i];
212            }
213
214            Ok(CubicSpline {
215                x: x.to_owned(),
216                y: y.to_owned(),
217                y2,
218            })
219        }
220
221        /// Interpolate at the given x value
222        pub fn interpolate(&self, x: f64) -> Result<f64, SplineError> {
223            // Find the right interval using binary search
224            let n = self.x.len();
225
226            // Handle out-of-bounds
227            if x < self.x[0] || x > self.x[n - 1] {
228                return Err(SplineError::OutOfBounds);
229            }
230
231            // Binary search to find the right interval
232            let mut low = 0;
233            let mut high = n - 1;
234
235            while high - low > 1 {
236                let mid = (high + low) / 2;
237                if self.x[mid] > x {
238                    high = mid;
239                } else {
240                    low = mid;
241                }
242            }
243
244            // Now x is between self.x[low] and self.x[high]
245            let h = self.x[high] - self.x[low];
246            if h == 0.0 {
247                return Err(SplineError::PointsTooClose);
248            }
249
250            // Cubic spline polynomial evaluation
251            let a = (self.x[high] - x) / h;
252            let b = (x - self.x[low]) / h;
253
254            // Evaluate the spline polynomial
255            let y = a * self.y[low]
256                + b * self.y[high]
257                + ((a * a * a - a) * self.y2[low] + (b * b * b - b) * self.y2[high]) * (h * h)
258                    / 6.0;
259
260            Ok(y)
261        }
262
263        /// Evaluate the spline with extrapolation outside bounds
264        /// This is a special method to handle cases where x is outside the range
265        pub fn eval_with_extrapolation(&self, x: f64) -> Result<f64, SplineError> {
266            let n = self.x.len();
267
268            // Handle in-bounds case
269            if x >= self.x[0] && x <= self.x[n - 1] {
270                return self.interpolate(x);
271            }
272
273            // If we're out of bounds, use linear extrapolation
274            if x < self.x[0] {
275                // Linear extrapolation below range using the first two points
276                let slope = (self.y[1] - self.y[0]) / (self.x[1] - self.x[0]);
277                Ok(self.y[0] + slope * (x - self.x[0]))
278            } else {
279                // Linear extrapolation above range using the last two points
280                let slope = (self.y[n - 1] - self.y[n - 2]) / (self.x[n - 1] - self.x[n - 2]);
281                Ok(self.y[n - 1] + slope * (x - self.x[n - 1]))
282            }
283        }
284    }
285    #[derive(Debug, Clone, PartialEq)]
286    pub enum SplineError {
287        /// Input arrays have different lengths
288        DifferentLengths,
289        /// Not enough points provided for interpolation
290        InsufficientPoints,
291        /// Input x values must be in strictly ascending order
292        UnsortedPoints,
293        /// Interpolation requested outside range of input data
294        OutOfBounds,
295        /// Points in the input data are too close together
296        PointsTooClose,
297        /// Numerical instability detected during calculation
298        NumericalError,
299    }
300
301    impl std::fmt::Display for SplineError {
302        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
303            match self {
304                SplineError::DifferentLengths => {
305                    write!(f, "Input arrays must have the same length")
306                }
307                SplineError::InsufficientPoints => {
308                    write!(f, "Need at least two points for interpolation")
309                }
310                SplineError::UnsortedPoints => {
311                    write!(f, "Input x values must be in strictly ascending order")
312                }
313                SplineError::OutOfBounds => write!(f, "Interpolation x value out of bounds"),
314                SplineError::PointsTooClose => write!(f, "Interpolation points too close"),
315                SplineError::NumericalError => write!(f, "Numerical error during computation"),
316            }
317        }
318    }

331    //! A simple quasi-random number generator based on the Halton sequence.
332    //! The Halton sequence is a sequence of numbers that are generated by
333    //! taking the fractional part of a number raised to a power. The Halton
334    //! sequence is defined for any positive integer `n`, and the power is
335    //! defined as `2^n`.
336
337    /// Halton State
338    pub struct Halton {
339        index: usize,
340        dimension: usize,
341    }
342
343    #[derive(Debug)]
344    pub enum HaltonError {
345        DimensionTooLarge,
346    }
347
348    impl std::fmt::Display for HaltonError {
349        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
350            match self {
351                HaltonError::DimensionTooLarge => {
352                    write!(f, "Halton sequence cannot have dimension greater than 1229")
353                }
354            }
355        }
356    }
357
358    impl std::error::Error for HaltonError {}
359
360    impl Halton {
361        pub fn new(dimension: usize) -> Result<Self, HaltonError> {
362            if dimension == 0 || dimension > PRIMES.len() {
363                return Err(HaltonError::DimensionTooLarge);
364            }
365            Ok(Halton {
366                index: 0,
367                dimension,
368            })
369        }
370
371        pub fn next_point(&mut self) -> Vec<f64> {
372            let mut point = Vec::with_capacity(self.dimension);
373            for &base in &PRIMES[..self.dimension] {
374                point.push(halton(self.index, base));
375            }
376            self.index += 1;
377            point
378        }
379    }
380
381    fn halton(mut index: usize, base: usize) -> f64 {
382        let mut result = 0.0;
383        let mut f = 1.0 / base as f64;
384        while index > 0 {
385            result += f * (index % base) as f64;
386            index /= base;
387            f /= base as f64;
388        }
389        result
390    }

493    /// `erf` calculates the error function at `x`.
494    pub fn erf(x: f64) -> f64 {
495        if x.is_nan() {
496            f64::NAN
497        } else if x >= 0.0 && x.is_infinite() {
498            1.0
499        } else if x <= 0.0 && x.is_infinite() {
500            -1.0
501        } else if x == 0.0 {
502            0.0
503        } else {
504            erf_impl(x, false)
505        }
506    }
507
508    /// `erf_inv` calculates the inverse error function
509    /// at `x`.
510    pub fn erf_inv(x: f64) -> f64 {
511        if x == 0.0 {
512            0.0
513        } else if x >= 1.0 {
514            f64::INFINITY
515        } else if x <= -1.0 {
516            f64::NEG_INFINITY
517        } else if x < 0.0 {
518            erf_inv_impl(-x, 1.0 + x, -1.0)
519        } else {
520            erf_inv_impl(x, 1.0 - x, 1.0)
521        }
522    }
523
524    /// `erfc` calculates the complementary error function
525    /// at `x`.
526    pub fn erfc(x: f64) -> f64 {
527        if x.is_nan() {
528            f64::NAN
529        } else if x == f64::INFINITY {
530            0.0
531        } else if x == f64::NEG_INFINITY {
532            2.0
533        } else {
534            erf_impl(x, true)
535        }
536    }
537
538    /// `erfc_inv` calculates the complementary inverse
539    /// error function at `x`.
540    pub fn erfc_inv(x: f64) -> f64 {
541        if x <= 0.0 {
542            f64::INFINITY
543        } else if x >= 2.0 {
544            f64::NEG_INFINITY
545        } else if x > 1.0 {
546            erf_inv_impl(-1.0 + x, 2.0 - x, -1.0)
547        } else {
548            erf_inv_impl(1.0 - x, x, 1.0)
549        }
550    }

1056    /// `erf_impl` computes the error function at `z`.
1057    /// If `inv` is true, `1 - erf` is calculated as opposed to `erf`
1058    fn erf_impl(z: f64, inv: bool) -> f64 {
1059        if z < 0.0 {
1060            if !inv {
1061                return -erf_impl(-z, false);
1062            }
1063            if z < -0.5 {
1064                return 2.0 - erf_impl(-z, true);
1065            }
1066            return 1.0 + erf_impl(-z, false);
1067        }
1068
1069        let result = if z < 0.5 {
1070            if z < 1e-10 {
1071                z * 1.125 + z * 0.003379167095512573896158903121545171688
1072            } else {
1073                z * 1.125 + z * aux::polynomial(z, ERF_IMPL_AN) / aux::polynomial(z, ERF_IMPL_AD)
1074            }
1075        } else if z < 110.0 {
1076            let (r, b) = if z < 0.75 {
1077                (
1078                    aux::polynomial(z - 0.5, ERF_IMPL_BN) / aux::polynomial(z - 0.5, ERF_IMPL_BD),
1079                    0.3440242112,
1080                )
1081            } else if z < 1.25 {
1082                (
1083                    aux::polynomial(z - 0.75, ERF_IMPL_CN) / aux::polynomial(z - 0.75, ERF_IMPL_CD),
1084                    0.419990927,
1085                )
1086            } else if z < 2.25 {
1087                (
1088                    aux::polynomial(z - 1.25, ERF_IMPL_DN) / aux::polynomial(z - 1.25, ERF_IMPL_DD),
1089                    0.4898625016,
1090                )
1091            } else if z < 3.5 {
1092                (
1093                    aux::polynomial(z - 2.25, ERF_IMPL_EN) / aux::polynomial(z - 2.25, ERF_IMPL_ED),
1094                    0.5317370892,
1095                )
1096            } else if z < 5.25 {
1097                (
1098                    aux::polynomial(z - 3.5, ERF_IMPL_FN) / aux::polynomial(z - 3.5, ERF_IMPL_FD),
1099                    0.5489973426,
1100                )
1101            } else if z < 8.0 {
1102                (
1103                    aux::polynomial(z - 5.25, ERF_IMPL_GN) / aux::polynomial(z - 5.25, ERF_IMPL_GD),
1104                    0.5571740866,
1105                )
1106            } else if z < 11.5 {
1107                (
1108                    aux::polynomial(z - 8.0, ERF_IMPL_HN) / aux::polynomial(z - 8.0, ERF_IMPL_HD),
1109                    0.5609807968,
1110                )
1111            } else if z < 17.0 {
1112                (
1113                    aux::polynomial(z - 11.5, ERF_IMPL_IN) / aux::polynomial(z - 11.5, ERF_IMPL_ID),
1114                    0.5626493692,
1115                )
1116            } else if z < 24.0 {
1117                (
1118                    aux::polynomial(z - 17.0, ERF_IMPL_JN) / aux::polynomial(z - 17.0, ERF_IMPL_JD),
1119                    0.5634598136,
1120                )
1121            } else if z < 38.0 {
1122                (
1123                    aux::polynomial(z - 24.0, ERF_IMPL_KN) / aux::polynomial(z - 24.0, ERF_IMPL_KD),
1124                    0.5638477802,

1141            };
1142            let g = (-z * z).exp() / z;
1143            g * b + g * r
1144        } else {
1145            0.0
1146        };
1147
1148        if inv && z >= 0.5 {

1157    // `erf_inv_impl` computes the inverse error function where
1158    // `p`,`q`, and `s` are the first, second, and third intermediate
1159    // parameters respectively
1160    fn erf_inv_impl(p: f64, q: f64, s: f64) -> f64 {
1161        let result = if p <= 0.5 {
1162            let y = 0.0891314744949340820313;
1163            let g = p * (p + 10.0);
1164            let r = aux::polynomial(p, ERF_INV_IMPL_AN) / aux::polynomial(p, ERF_INV_IMPL_AD);
1165            g * y + g * r
1166        } else if q >= 0.25 {
1167            let y = 2.249481201171875;
1168            let g = (-2.0 * q.ln()).sqrt();
1169            let xs = q - 0.25;
1170            let r = aux::polynomial(xs, ERF_INV_IMPL_BN) / aux::polynomial(xs, ERF_INV_IMPL_BD);
1171            g / (y + r)
1172        } else {
1173            let x = (-q.ln()).sqrt();
1174            if x < 3.0 {
1175                let y = 0.807220458984375;
1176                let xs = x - 1.125;
1177                let r = aux::polynomial(xs, ERF_INV_IMPL_CN) / aux::polynomial(xs, ERF_INV_IMPL_CD);
1178                y * x + r * x
1179            } else if x < 6.0 {
1180                let y = 0.93995571136474609375;
1181                let xs = x - 3.0;
1182                let r = aux::polynomial(xs, ERF_INV_IMPL_DN) / aux::polynomial(xs, ERF_INV_IMPL_DD);
1183                y * x + r * x
1184            } else if x < 18.0 {
1185                let y = 0.98362827301025390625;
1186                let xs = x - 6.0;
1187                let r = aux::polynomial(xs, ERF_INV_IMPL_EN) / aux::polynomial(xs, ERF_INV_IMPL_ED);
1188                y * x + r * x
1189            } else if x < 44.0 {
1190                let y = 0.99714565277099609375;
1191                let xs = x - 18.0;
1192                let r = aux::polynomial(xs, ERF_INV_IMPL_FN) / aux::polynomial(xs, ERF_INV_IMPL_FD);
1193                y * x + r * x
1194            } else {
1195                let y = 0.99941349029541015625;
1196                let xs = x - 44.0;
1197                let r = aux::polynomial(xs, ERF_INV_IMPL_GN) / aux::polynomial(xs, ERF_INV_IMPL_GD);
1198                y * x + r * x
1199            }
1200        };
1201        s * result
1202    }

1222    /// evaluates a polynomial at `z` where `coeff` are the coeffecients
1223    /// to a polynomial of order `k` where `k` is the length of `coeff` and the
1224    /// coeffecient
1225    /// to the `k`th power is the `k`th element in coeff. E.g. [3,-1,2] equates to
1226    /// `2z^2 - z + 3`
1227    ///
1228    /// # Remarks
1229    ///
1230    /// Returns 0 for a 0 length coefficient slice
1231    pub fn polynomial(z: f64, coeff: &[f64]) -> f64 {
1232        let n = coeff.len();
1233        if n == 0 {
1234            return 0.0;
1235        }
1236
1237        let mut sum = *coeff.last().unwrap();
1238        for c in coeff[0..n - 1].iter().rev() {
1239            sum = *c + z * sum;
1240        }
1241        sum
1242    }
