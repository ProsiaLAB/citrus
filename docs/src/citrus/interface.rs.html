21#[deprecated = "use `python`-defined function instead"]
22pub fn density(x: f64, y: f64, z: f64) -> f64 {
23    let r_min = 0.7 * cc::AU_SI;
24
25    // Calculate radial distance from origin
26    let r = x * x + y * y + z * z;
27
28    // Calculate a spherical power-law density profile
29    let r_to_use = if r > r_min { r } else { r_min };
30
31    // Get density after converting to SI units
32    1.5e6 * (r_to_use / (300.0 * cc::AU_SI)).powf(-1.5) * 1e6
33}

35#[deprecated = "use `python`-defined function instead"]
36fn temperature(x: f64, y: f64, z: f64) -> f64 {
37    let mut x0: usize = 0;
38
39    let temp = [
40        [
41            2.0e13, 5.0e13, 8.0e13, 1.1e14, 1.4e14, 1.7e14, 2.0e14, 2.3e14, 2.6e14, 2.9e14,
42        ],
43        [
44            44.777, 31.037, 25.718, 22.642, 20.560, 19.023, 17.826, 16.857, 16.050, 15.364,
45        ],
46    ];
47
48    // Calculate radial distance from origin
49    let r = (x * x + y * y + z * z).sqrt();
50
51    // Linear interpolation for temperature input
52    if r > temp[0][0] && r < temp[0][9] {
53        for i in 0..9 {
54            if r > temp[0][i] && r < temp[0][i + 1] {
55                x0 = i;
56            }
57        }
58    }
59    if r < temp[0][0] {
60        temp[1][0]
61    } else if r > temp[0][9] {
62        temp[1][9]
63    } else {
64        temp[1][x0]
65            + (r - temp[0][x0]) * (temp[1][x0 + 1] - temp[1][x0]) / (temp[0][x0 + 1] - temp[0][x0])
66    }
67}

85pub fn velocity(x: f64, y: f64, z: f64) -> RVector {
86    let r_min = 0.1 * cc::AU_SI;
87
88    // Calculate radial distance from origin
89    let r = (x * x + y * y + z * z).sqrt();
90
91    let r_to_use = if r > r_min { r } else { r_min };
92
93    // Free-fall velocity in the radial direction onto a central mass of 1 solar mass
94    let free_fall_velocity = (2.0 * cc::GRAVITATIONAL_CONST_SI * 1.989e30 / r_to_use).sqrt();
95    RVector::from_vec(vec![
96        -x * free_fall_velocity / r_to_use,
97        -y * free_fall_velocity / r_to_use,
98        -z * free_fall_velocity / r_to_use,
99    ])
