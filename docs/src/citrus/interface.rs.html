24#[deprecated = "use `python`-defined function instead"]
25pub fn density(x: f64, y: f64, z: f64) -> f64 {
26    let r_min = 0.7 * cc::AU_SI;
27
28    // Calculate radial distance from origin
29    let r = x * x + y * y + z * z;
30
31    // Calculate a spherical power-law density profile
32    let r_to_use = if r > r_min { r } else { r_min };
33
34    // Get density after converting to SI units
35    1.5e6 * (r_to_use / (300.0 * cc::AU_SI)).powf(-1.5) * 1e6
36}

38#[deprecated = "use `python`-defined function instead"]
39fn temperature(x: f64, y: f64, z: f64) -> f64 {
40    let mut x0: usize = 0;
41
42    let temp = [
43        [
44            2.0e13, 5.0e13, 8.0e13, 1.1e14, 1.4e14, 1.7e14, 2.0e14, 2.3e14, 2.6e14, 2.9e14,
45        ],
46        [
47            44.777, 31.037, 25.718, 22.642, 20.560, 19.023, 17.826, 16.857, 16.050, 15.364,
48        ],
49    ];
50
51    // Calculate radial distance from origin
52    let r = (x * x + y * y + z * z).sqrt();
53
54    // Linear interpolation for temperature input
55    if r > temp[0][0] && r < temp[0][9] {
56        for i in 0..9 {
57            if r > temp[0][i] && r < temp[0][i + 1] {
58                x0 = i;
59            }
60        }
61    }
62    if r < temp[0][0] {
63        temp[1][0]
64    } else if r > temp[0][9] {
65        temp[1][9]
66    } else {
67        temp[1][x0]
68            + (r - temp[0][x0]) * (temp[1][x0 + 1] - temp[1][x0]) / (temp[0][x0 + 1] - temp[0][x0])
69    }
70}

87#[deprecated = "use `python`-defined function instead"]
88pub fn velocity(x: f64, y: f64, z: f64) -> RVector {
89    let r_min = 0.1 * cc::AU_SI;
90
91    // Calculate radial distance from origin
92    let r = (x * x + y * y + z * z).sqrt();
93
94    let r_to_use = if r > r_min { r } else { r_min };
95
96    // Free-fall velocity in the radial direction onto a central mass of 1 solar mass
97    let free_fall_velocity = (2.0 * cc::GRAVITATIONAL_CONST_SI * 1.989e30 / r_to_use).sqrt();
98    RVector::from_vec(vec![
99        -x * free_fall_velocity / r_to_use,
100        -y * free_fall_velocity / r_to_use,
101        -z * free_fall_velocity / r_to_use,
102    ])
