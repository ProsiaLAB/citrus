9pub struct PythonModel {
10    py_module: Py<PyModule>,
11}
12
13impl PythonModel {
14    /// Creates a new PythonModel by loading a Python file.
15    ///
16    /// The `file_name` can be just the name (e.g., "model.py") if the file is
17    /// in the current working directory or a directory already on Python's sys.path.
18    /// It can also be a relative or absolute path.
19    pub fn new(file_name: &str) -> PyResult<Self> {
20        Python::with_gil(|py| {
21            let full_path = PathBuf::from(file_name);
22
23            // Extract the module name (file name without extension)
24            let module_name_str = full_path.file_stem()
25                .ok_or_else(|| PyErr::new::<pyo3::exceptions::PyValueError, _>(
26                    format!("Invalid model file name: '{}' has no file name component.", file_name)
27                ))?
28                .to_str()
29                .ok_or_else(|| PyErr::new::<pyo3::exceptions::PyValueError, _>(
30                    format!("Invalid model file name: '{}' is not valid UTF-8.", file_name)
31                ))?;
32
33            // Get the directory containing the file.
34            // If `file_name` is just "model.py", parent() will be Some("").
35            // If `file_name` is "subdir/model.py", parent() will be Some("subdir").
36            let parent_dir_str = full_path.parent()
37                .ok_or_else(|| PyErr::new::<pyo3::exceptions::PyValueError, _>(
38                    format!("Invalid model file name: '{}' has no parent directory component.", file_name)
39                ))?
40                .to_str()
41                .ok_or_else(|| PyErr::new::<pyo3::exceptions::PyValueError, _>(
42                    format!("Invalid model file path: parent directory of '{}' is not valid UTF-8.", file_name)
43                ))?;
44
45            // Add the directory to Python's sys.path so the module can be found.
46            // An empty string for parent_dir_str correctly represents the current working directory.
47            if !parent_dir_str.is_empty() {
48                let sys = py.import("sys")?;
49                let sys_path = sys.getattr("path")?;
50                // Insert at the beginning to prioritize our custom path
51                sys_path.call_method1("insert", (0, parent_dir_str))?;
52            }
53            // If parent_dir_str is empty (""), it implies the current directory,
54            // which is usually already on sys.path, so no explicit insert is needed.
55
56            // Import the module by its name
57            let module = PyModule::import(py, module_name_str)?;
58
59            Ok(PythonModel {
60                py_module: module.into(),
61            })
62        })
63    }
64
65    fn call_f64(&self, name: &str, x: f64, y: f64, z: f64, fallback: f64) -> f64 {
66        Python::with_gil(|py| {
67            let m = self.py_module.bind(py);
68            match m.getattr(name) {
69                Ok(f) => f.call1((x, y, z))
70                          // Use a closure to call .extract() on the Bound<'py, PyAny> result
71                          .and_then(|result_bound_pyany| result_bound_pyany.extract())
72                          .unwrap_or_else(|e| {
73                              eprintln!("Error extracting f64 or calling Python function '{}': {}", name, e);
74                              fallback
75                          }),
76                Err(_) => fallback,
77            }
78        })
79    }
80}
